Using two ngIf
<div *ngIf="!blueControl.value" class="col-auto">
  <div class="box red"></div>
</div>
        
<div *ngIf="blueControl.value" class="col-auto">
  <div class="box blue"></div>
</div>
Using the new @if
@if(!blueControl.value) {
} @else {
} 
@if(!blueControl.value) {
  <div class="col-auto">
    <div class="box red"></div>
  </div>
}
        
@else {          
<div class="col-auto">
  <div class="box blue"></div>
</div>
}
Using else
 
<ng-template #useRed>
  <div class=" box red"></div>
</ng-template>
        
<div class="col-auto">
  <div *ngIf="blueControl.value; else useRed" 
    class="box blue"></div>
</div>
Using then/else
 
<ng-template #useRed>
  <div class="box red"></div>
</ng-template>
        
<ng-template #useBlue>
  <div class="box blue"></div>
</ng-template>
        
<div class="col-auto">
  <ng-container *ngIf="blueControl.value; 
    then useBlue else useRed"></ng-container>
</div>

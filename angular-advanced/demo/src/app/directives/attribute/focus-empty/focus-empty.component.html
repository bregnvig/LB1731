                      export class FocusEmptyDirective {
                      
                        constructor(elementRef: ElementRef) {}
                      
                        @HostListener('submit')
                        onFormSubmit() {
                          const emptyControl = this.elementRef.nativeElement.querySelector('input.ng-invalid:not([value]):not([value=""])');
                          emptyControl && emptyControl.focus();
                        }
